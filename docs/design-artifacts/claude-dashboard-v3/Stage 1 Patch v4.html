function renderChoice(opts: {
  action: "import-file" | "load-excel-demo" | "load-sample-csv";
  label: string; desc: string; meta: string;
  primary?: boolean;
}): string {
  const cls = opts.primary ? "bw-choice bw-choice--primary" : "bw-choice";
  // `action` is a closed string-literal union so it doesn't need escaping,
  // but label/desc/meta may be (or become) data-driven — escape them all.
  return `
    <button type="button" class="${cls}" data-bw-action="${opts.action}">
      <span class="bw-choice__label">${escapeHtml(opts.label)}</span>
      <span class="bw-choice__desc">${escapeHtml(opts.desc)}</span>
      <span class="bw-choice__meta">${escapeHtml(opts.meta)}</span>
    </button>
  `;
}

function renderSampleAltLinks(samples: readonly SampleDataset[]): string {
  if (samples.length <= 1) return "";
  const links = samples.map((s, i) => `
    ${i > 0 ? '<span class="bw-samples__sep" aria-hidden="true"> · </span>' : ""}
    <button type="button" class="bw-samples__link"
            data-bw-action="load-sample-csv"
            data-bw-sample-path="${escapeHtml(s.path)}">${escapeHtml(s.label)}</button>
  `).join("");
  return `<p class="bw-samples">or load: ${links}</p>`;
}

// src/ui/pre-import-actions.ts — new file

// Existing sample / Northstar flows are async in main.ts, so handler types must
// allow either signature. The dispatcher uses `void` so floating promises are
// explicit and lint-clean (no-floating-promises stays green).
export type PreImportActionHandlers = {
  openFilePicker:     () => void | Promise<void>;
  loadNorthstarDemo:  () => void | Promise<void>;
  loadSelectedSample: (samplePath?: string) => void | Promise<void>;
};

export type BindPreImportActionsOpts = PreImportActionHandlers & {
  root: HTMLElement;
};

/**
 * Delegate clicks on data-bw-action elements inside <root> to the supplied
 * handlers. Returns an unbind function. The listener is attached to <root>,
 * which is stable across innerHTML swaps of its descendants — no rebinding
 * needed when the panel is re-painted by Clear or error recovery.
 */
export function bindPreImportActions(opts: BindPreImportActionsOpts): () => void {
  const handler = (e: Event) => {
    const btn = (e.target as HTMLElement).closest<HTMLElement>("[data-bw-action]");
    if (!btn) return;
    if (btn.hasAttribute("disabled") || btn.getAttribute("aria-disabled") === "true") return;
    switch (btn.getAttribute("data-bw-action")) {
      case "import-file":     void opts.openFilePicker(); break;
      case "load-excel-demo": void opts.loadNorthstarDemo(); break;
      case "load-sample-csv": void opts.loadSelectedSample(btn.dataset.bwSamplePath); break;
    }
  };

  opts.root.addEventListener("click", handler);
  return () => opts.root.removeEventListener("click", handler);
}

Bind once to a root that contains both the app-bar load button and #results. The repo’s app root element (commonly document.body or the outer .app wrapper) is the natural choice. Listener attached to a parent element survives innerHTML replacement of any descendant, so re-paints don’t require rebinding.

Note · The third case forwards data-bw-sample-path as an optional argument. When the user clicks the big “Load sample CSV” choice (no data-bw-sample-path), loadSelectedSample(undefined) falls back to the default behavior (first sample). When a specific alt-link is clicked, the sample path is forwarded — paths are the stable handle in SAMPLE_DATASETS, not ids.
