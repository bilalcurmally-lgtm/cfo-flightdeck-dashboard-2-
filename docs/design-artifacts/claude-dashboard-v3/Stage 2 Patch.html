01 / Helper

src/finance/cockpit-kpis.ts — full file.Pure. Composes FinanceSummary; never recomputes the totals the finance layer already owns.

// src/finance/cockpit-kpis.ts — new file

import type { FinanceSummary } from "./summary";
import type { TransactionRecord } from "./types";

export type RunwayTone = "healthy" | "watch" | "tight" | "unknown";

export type ReviewBreakdown = {
  rejected:   number;
  duplicates: number;
  transfers:  number;
  total:      number;
};

export type CockpitViewModel = {
  // Pass-throughs from FinanceSummary (no recomputation):
  revenue:               number;
  outflow:               number;
  netCash:               number;
  runwayMonths:          number | null;
  averageMonthlyOutflow: number;

  // Cockpit-specific derived values:
  inflowCount:           number;
  outflowCount:          number;
  runwayTone:            RunwayTone;
  review:                ReviewBreakdown;
  hasRows:               boolean;
};

export type DeriveCockpitInput = {
  summary:      FinanceSummary;
  records:      readonly TransactionRecord[];
  rejectedRows: readonly unknown[];
};

function classifyRunway(months: number | null): RunwayTone {
  if (months === null || !Number.isFinite(months)) return "unknown";
  if (months >= 9) return "healthy";
  if (months >= 3) return "watch";
  return "tight";
}

export function deriveCockpit(input: DeriveCockpitInput): CockpitViewModel {
  const { summary, records, rejectedRows } = input;
  const { diagnostics, cashHealth } = summary;

  let inflowCount = 0;
  let outflowCount = 0;
  for (const r of records) {
    if (r.flow === "revenue") inflowCount++;
    else if (r.flow === "outflow") outflowCount++;
  }

  const rejected   = rejectedRows.length;
  const duplicates = diagnostics.duplicateGroups.length;
  const transfers  = diagnostics.transferCandidates.length;

  const review: ReviewBreakdown = {
    rejected,
    duplicates,
    transfers,
    total: rejected + duplicates + transfers,
  };

  return {
    revenue:               summary.revenue,
    outflow:               summary.outflow,
    netCash:               summary.netCash,
    runwayMonths:          cashHealth.runwayMonths,
    averageMonthlyOutflow: cashHealth.averageMonthlyOutflow,
    inflowCount,
    outflowCount,
    runwayTone:            classifyRunway(cashHealth.runwayMonths),
    review,
    hasRows:               summary.transactionCount > 0,
  };
}
One thing to verify at land time · if FinanceSummary.diagnostics uses different field names than duplicateGroups / transferCandidates, swap those two property reads to the actual names and update the test fixture in §02 to match. Everything else compiles against the real types as-is.

// src/finance/cockpit-kpis.test.ts — new file

import { describe, test, expect } from "vitest";
import { deriveCockpit } from "./cockpit-kpis";
import type { FinanceSummary } from "./summary";
import type { TransactionRecord } from "./types";

// Minimal record + summary builders. Cast through `unknown` so the test
// stays robust to additional required fields on the real types — we only
// touch the fields the cockpit reads.
const rec = (flow: "revenue" | "outflow", amount: number): TransactionRecord =>
  ({ flow, amount, signedNet: flow === "revenue" ? amount : -amount } as unknown as TransactionRecord);

const baseSummary: FinanceSummary = {
  revenue: 1000,
  outflow: 600,
  netCash: 400,
  transactionCount: 5,
  cashHealth: { runwayMonths: 6, averageMonthlyOutflow: 200 },
  diagnostics: { duplicateGroups: [], transferCandidates: [] },
  warnings: [],
} as unknown as FinanceSummary;

describe("deriveCockpit", () => {

  test("T1 \u00b7 counts inflow vs outflow by record.flow, never by sign", () => {
    const vm = deriveCockpit({
      summary: baseSummary,
      records: [rec("revenue", 100), rec("revenue", 200), rec("outflow", 50)],
      rejectedRows: [],
    });
    expect(vm.inflowCount).toBe(2);
    expect(vm.outflowCount).toBe(1);
  });

  test("T2 \u00b7 passes FinanceSummary numbers through verbatim", () => {
    const vm = deriveCockpit({ summary: baseSummary, records: [], rejectedRows: [] });
    expect(vm.revenue).toBe(1000);
    expect(vm.outflow).toBe(600);
    expect(vm.netCash).toBe(400);
    expect(vm.runwayMonths).toBe(6);
    expect(vm.averageMonthlyOutflow).toBe(200);
  });

  test("T3 \u00b7 runway tone thresholds at 3 and 9 months; null/Infinity \u2192 unknown", () => {
    const tone = (m: number | null) => deriveCockpit({
      summary: { ...baseSummary, cashHealth: { ...baseSummary.cashHealth, runwayMonths: m } } as unknown as FinanceSummary,
      records: [],
      rejectedRows: [],
    }).runwayTone;

    expect(tone(12)).toBe("healthy");
    expect(tone(9)).toBe("healthy");
    expect(tone(5)).toBe("watch");
    expect(tone(3)).toBe("watch");
    expect(tone(2.9)).toBe("tight");
    expect(tone(null)).toBe("unknown");
    expect(tone(Number.POSITIVE_INFINITY)).toBe("unknown");
  });

  test("T4 \u00b7 review total = rejected + duplicateGroups + transferCandidates", () => {
    const vm = deriveCockpit({
      summary: {
        ...baseSummary,
        diagnostics: { duplicateGroups: [{}, {}], transferCandidates: [{}] },
      } as unknown as FinanceSummary,
      records: [],
      rejectedRows: [{}, {}, {}, {}],
    });
    expect(vm.review).toEqual({ rejected: 4, duplicates: 2, transfers: 1, total: 7 });
  });

  test("T5 \u00b7 hasRows mirrors summary.transactionCount > 0", () => {
    const empty = deriveCockpit({
      summary: { ...baseSummary, transactionCount: 0 } as unknown as FinanceSummary,
      records: [],
      rejectedRows: [],
    });
    const full = deriveCockpit({ summary: baseSummary, records: [], rejectedRows: [] });
    expect(empty.hasRows).toBe(false);
    expect(full.hasRows).toBe(true);
  });

});
03 / Renderer

src/ui/dashboard-cockpit.ts — full file.Returns an HTML string. Formatters injected. escapeHtml on every dynamic value.

// src/ui/dashboard-cockpit.ts — new file

import { escapeHtml } from "./html";
import type {
  CockpitViewModel,
  ReviewBreakdown,
} from "../finance/cockpit-kpis";

export type CockpitFormatters = {
  formatMoney:  (n: number) => string;
  formatRunway: (months: number | null) => string;
};

function signedMoney(n: number, fmt: (x: number) => string): string {
  // formatMoney already handles negative; we only prepend "+" for positives.
  return n > 0 ? `+${fmt(n)}` : fmt(n);
}

function reviewMeta(r: ReviewBreakdown): string {
  const parts: string[] = [];
  if (r.rejected)   parts.push(`${r.rejected} rejected`);
  if (r.duplicates) parts.push(`${r.duplicates} dupe${r.duplicates === 1 ? "" : "s"}`);
  if (r.transfers)  parts.push(`${r.transfers} transfer${r.transfers === 1 ? "" : "s"}`);
  return parts.join(" \u00b7 ");
}

type TileOpts = {
  label:      string;
  value:      string;
  meta?:      string;
  modifier?:  "primary" | "anchor" | "tight" | "review";
  metaClass?: "meta" | "tone"; // "tone" is used only on the Runway tile
};

function renderTile(opts: TileOpts): string {
  const cls = ["bw-kpi", opts.modifier && `bw-kpi--${opts.modifier}`]
    .filter(Boolean)
    .join(" ");
  const metaCls = opts.metaClass ?? "meta";
  const meta = opts.meta
    ? `<span class="bw-kpi__${metaCls}">${escapeHtml(opts.meta)}</span>`
    : "";
  return `
    <div class="${cls}">
      <span class="bw-kpi__label">${escapeHtml(opts.label)}</span>
      <span class="bw-kpi__value">${escapeHtml(opts.value)}</span>
      ${meta}
    </div>
  `;
}

export function renderCockpitStrip(
  vm: CockpitViewModel,
  fmt: CockpitFormatters,
): string {
  const showReview = vm.review.total > 0;
  const empty       = !vm.hasRows;
  const dash        = "\u2014";

  const runwayModifier: "anchor" | "tight" =
    vm.runwayTone === "tight" ? "tight" : "anchor";

  const burnMeta = vm.averageMonthlyOutflow > 0
    ? `${vm.runwayTone} \u00b7 burn ${fmt.formatMoney(vm.averageMonthlyOutflow)}/mo`
    : vm.runwayTone === "unknown"
      ? "runway unavailable"
      : "no recorded burn";

  const tiles = [
    renderTile({
      label: "Revenue",
      value: empty ? dash : fmt.formatMoney(vm.revenue),
      meta:  empty ? "no rows in current filter" : `${vm.inflowCount} rows`,
    }),
    renderTile({
      label: "Outflow",
      value: empty ? dash : fmt.formatMoney(vm.outflow),
      meta:  empty ? "no rows in current filter" : `${vm.outflowCount} rows`,
    }),
    renderTile({
      label: "Net cash",
      value: empty ? dash : signedMoney(vm.netCash, fmt.formatMoney),
      meta:  empty ? undefined : `${vm.inflowCount + vm.outflowCount} rows`,
      modifier: "primary",
    }),
    renderTile({
      label: "Runway",
      value: empty ? dash : fmt.formatRunway(vm.runwayMonths),
      meta:  empty ? undefined : burnMeta,
      modifier:  runwayModifier,
      metaClass: "tone",
    }),
    showReview ? renderTile({
      label: "Needs review",
      value: String(vm.review.total),
      meta:  reviewMeta(vm.review),
      modifier: "review",
    }) : "",
  ].join("");

  const rootCls = showReview ? "bw-cockpit" : "bw-cockpit bw-cockpit--4";
  return `<section class="${rootCls}" role="group" aria-label="Cockpit summary">${tiles}</section>`;
}
Escape coverage · label, value, and meta all flow through escapeHtml. modifier and metaClass are closed string-literal unions, safe by construction.

// src/ui/dashboard-cockpit.test.ts — new file

import { describe, test, expect } from "vitest";
import { renderCockpitStrip } from "./dashboard-cockpit";
import type { CockpitViewModel } from "../finance/cockpit-kpis";

const fmt = {
  formatMoney:  (n: number) => `$${n.toLocaleString("en-US")}`,
  formatRunway: (m: number | null) => (m === null ? "\u2014" : `${m} mo`),
};

const baseVm: CockpitViewModel = {
  revenue: 48210, outflow: 36540, netCash: 11670,
  runwayMonths: 6, averageMonthlyOutflow: 5220,
  inflowCount: 142, outflowCount: 318,
  runwayTone: "watch",
  review: { rejected: 4, duplicates: 2, transfers: 1, total: 7 },
  hasRows: true,
};

const countTiles = (html: string) => (html.match(/class="bw-kpi(?![_-])[^"]*"/g) ?? []).length;

describe("renderCockpitStrip", () => {

  test("T6 \u00b7 renders five tiles and the review tile carries the total", () => {
    const html = renderCockpitStrip(baseVm, fmt);
    expect(countTiles(html)).toBe(5);
    expect(html).toMatch(/bw-kpi--review/);
    expect(html).toMatch(/>7<\/span>/);
    expect(html).not.toMatch(/bw-cockpit--4/);
  });

  test("T7 \u00b7 review tile is omitted at zero, root carries --4 modifier", () => {
    const html = renderCockpitStrip(
      { ...baseVm, review: { rejected: 0, duplicates: 0, transfers: 0, total: 0 } },
      fmt,
    );
    expect(countTiles(html)).toBe(4);
    expect(html).not.toMatch(/bw-kpi--review/);
    expect(html).toMatch(/bw-cockpit--4/);
  });

  test("T8 \u00b7 tight runway swaps the anchor modifier for --tight", () => {
    const html = renderCockpitStrip(
      { ...baseVm, runwayMonths: 2, runwayTone: "tight" },
      fmt,
    );
    expect(html).toMatch(/bw-kpi--tight/);
    expect(html).not.toMatch(/bw-kpi--anchor/);
  });

  test("T9 \u00b7 empty filter \u2192 dashes + 'no rows' meta, never crashes", () => {
    const html = renderCockpitStrip(
      {
        ...baseVm,
        hasRows: false,
        review: { rejected: 0, duplicates: 0, transfers: 0, total: 0 },
      },
      fmt,
    );
    expect(html).toMatch(/\u2014/);
    expect(html).toMatch(/no rows in current filter/);
    expect(countTiles(html)).toBe(4);
  });

});
Tile counter · the regex class="bw-kpi(?![_-])[^"]*" deliberately excludes bw-kpi__label, bw-kpi__value, bw-kpi__meta, bw-kpi__tone, and bw-kpi--* modifier-only repeats. It counts root tile elements, which is the only number we care about.
